function getPermissionsTable(user) {
  let permissions = [];
  switch(user.role) {
    case 'sysadmin':
      permissions.push('sysadmin');
    case 'admin':
      permissions.push('admin');
    case 'user':
      permissions.push('user');
    default:
      permissions.push('anonymous');
      break;
  }
  return permissions;
}
